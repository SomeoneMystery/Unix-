#Array reverse

i=0
while [ $i -lt $length ]
do
    echo "${greetings[$i]}"
    i=$(( $i+1 ))
done

i=${#greetings[@]}
while [ $i -ge 0 ]
do
    echo "${greetings[$i]}"
    i=$(( i-1 ))
